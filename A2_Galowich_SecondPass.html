## linear regression - one variable (# of windows)
filler = [0]*len(windows)
oneVar_x = np.vstack((windows,filler))
reg_oneVar = LinearRegression().fit(oneVar_x.T,total_btu)
oneVar_Rsq = reg_oneVar.score(oneVar_x.T,total_btu)
oneVar_coeff = reg_oneVar.coef_
oneVar_yint = reg_oneVar.intercept_

print('R Squared Value = {}'.format(oneVar_Rsq))
print('Coefficient: number of windows = {}'.format(oneVar_coeff[0]))
print('y-intercept = {}'.format(oneVar_yint))

## linear regression - multivariate 
reg_fiveVar = LinearRegression().fit(drivers.T,total_btu)
fiveVar_Rsq = reg_fiveVar.score(drivers.T,total_btu)
fiveVar_coeff = reg_fiveVar.coef_
fiveVar_yint = reg_fiveVar.intercept_
fiveVar_adjRsq = 1-(1-fiveVar_Rsq)*((drivers.shape[1]-1)/
                                    (drivers.shape[1]-drivers.shape[0]-1))
